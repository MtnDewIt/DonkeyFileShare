This file serves as a reference to what a customization file contains

Example usage:

consumables[<consumable_index(0 - 3)>]: <consumable_name>
consumables[0]: empty
consumables[1]: empty
consumables[2]: empty
consumables[3]: empty

colors[<color_type_name>]: #RGB
colors[primary]: #FF0000
colors[secondary]: #0000FF
colors[visor]: #0000FF
colors[lights]: #00F00F
colors[holo]: #0F0F00

armor[<armor_region_name>]: <armor_name>
armor[helmet]: base
armor[chest]: base
armor[shoulders]: base
armor[arms]: base
armor[legs]: base
armor[acc]: flashlight
armor[pelvis]: base
